📱 MONGODB ATLAS SETUP GUIDE
================================

🎯 YES - IT WILL STORE MILLIONS OF DATA!
MongoDB is designed for this. Free tier: 512MB
Paid tier: UNLIMITED data storage and scaling

⏱️ Time needed: 5-10 minutes

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 1: CREATE FREE MONGODB ATLAS ACCOUNT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Open: https://www.mongodb.com/cloud/atlas
2. Click "Sign up free" or "Create free account"
3. Enter email/password
4. Click email verification link

✓ Account created!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 2: CREATE A CLUSTER
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Login to MongoDB Atlas
2. Click "Create" → "Build a cluster"
3. Choose "FREE" tier (M0)
4. Select region close to you (e.g., ap-southeast-1 for India)
5. Give it a name: "dinegrid"
6. Click "Create"
7. Wait 2-5 minutes for cluster to deploy

✓ Cluster created!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 3: CREATE DATABASE USER
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Left side → "Security" → "Database Access"
2. Click "Add New Database User"
3. Username: dinegrid
4. Password: Create a strong one (or use auto-generated)
5. Click "Add User"

✓ User created!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 4: ALLOW NETWORK ACCESS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Left side → "Security" → "Network Access"
2. Click "Add IP Address"
3. Click "Allow Access from Anywhere" 
   (For development only! Use specific IPs in production)
4. Click "Confirm"

✓ Network access configured!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 5: GET CONNECTION STRING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Click "Clusters" on left side
2. Click "Connect" button on your cluster
3. Choose "Drivers"
4. Select "Node.js" driver version
5. Copy the connection string (looks like):

   mongodb+srv://dinegrid:PASSWORD@cluster0.mongodb.net/dinegrid?retryWrites=true&w=majority

⚠️ Replace PASSWORD with your actual database password!

✓ Connection string ready!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 6: UPDATE server-db.js
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Open: c:\Users\hazar\Desktop\mvp\server-db.js

Find line 17:
const MONGODB_URI = process.env.MONGODB_URI || 'mongodb+srv://user:password@cluster0.mongodb.net/dinegrid?retryWrites=true&w=majority';

Replace with your connection string:
const MONGODB_URI = 'mongodb+srv://dinegrid:YOUR_PASSWORD@cluster-xxxx.mongodb.net/dinegrid?retryWrites=true&w=majority';

✓ Configuration updated!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 7: START THE SERVER
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

cd c:\Users\hazar\Desktop\mvp
